

class Solution {
    public long[] optimalArray(int n,int a[])
    {
        long[] ans=new long[n];
        
        ans[0]=0;
        for(int i=1;i<n;i++)
        {
            int index=i/2;
            ans[i]=ans[i-1]+(a[i]-a[i/2]);
        }
        
            return ans;
    }
}
